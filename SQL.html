WITH global_bolivar AS (
  SELECT Numero, Departamento
  FROM `global.clientes`
  WHERE Departamento = 'Bolivar'
),
bolivar_compras AS (
  SELECT Numero, Departamento, producto
  FROM Global_bolivar
  INNER JOIN `global.compras`
  ON Numero = cliente
),
bolivar_prod AS (
  SELECT Codigo, gprod.Producto AS Nombre_Prod
  FROM `global.productos` gprod
  INNER JOIN bolivar_compras bc
  ON bc.producto = gprod.Codigo
)
SELECT SUM(1) AS Total, bolivar_prod.Nombre_Prod
FROM bolivar_prod
GROUP BY Codigo, bolivar_prod.Nombre_Prod
ORDER BY Total DESC

WITH rango_precio AS(
  SELECT Precio, Codigo
  FROM `global.productos`
  WHERE Precio < 20000 AND Precio > 8000
),

indx_compras AS (
  SELECT Precio, Codigo, Cliente
  FROM rango_precio
  INNER JOIN `global.compras`
  ON Codigo = producto
),

conteo_clientes AS (
  SELECT Numero
  FROM `global.clientes`
  INNER JOIN indx_compras
  ON Cliente = Numero
  GROUP BY Numero
)

SELECT COUNT(*)
FROM conteo_clientes

WITH TotalCompras_Cliente AS (
SELECT Nombre_1, Nombre_2, Apellidos, COUNT (producto) AS nt_compras, Numero
FROM `global.clientes` gcl
INNER JOIN `global.compras` gcc
ON gcl.Numero = gcc.cliente
GROUP BY Nombre_1, Nombre_2, Apellidos, Numero
ORDER BY nt_compras DESC
),

compras AS (
  SELECT gcc.*, Precio
  FROM `global.compras`gcc
  INNER JOIN `global.productos`
  ON gcc.Producto = Codigo
)
SELECT Nombre_1, Nombre_2, Apellidos, TotalCompras_Cliente.nt_compras, SUM (Precio) AS VT_Compras
FROM compras
INNER JOIN TotalCompras_Cliente
ON Cliente = Numero
GROUP BY Nombre_1, Nombre_2, Apellidos, TotalCompras_Cliente.nt_compras
ORDER BY TotalCompras_Cliente.nt_compras DESC

WITH UGeografica AS (
  SELECT Departamento, producto, COUNT(producto) AS Comprados_Prod
  FROM `global.clientes`
  INNER JOIN `global.compras`gcc
  ON Cliente = Numero
  GROUP BY Departamento, producto
  ORDER BY Departamento, comprados_prod DESC
),

prod_depa AS(
SELECT Departamento, MAX(UGeografica.Comprados_Prod) AS QMProd_Comprado
FROM UGeografica
GROUP BY Departamento
ORDER BY QMPROD_COMPRADO DESC
),

cod_prod AS (
SELECT Prod_Depa.Departamento, producto, UGeografica.Comprados_Prod
FROM UGeografica
INNER JOIN prod_depa
ON (Prod_Depa.Departamento = UGeografica.Departamento AND UGeografica.Comprados_Prod = prod_depa.QMProd_Comprado)
)

SELECT Departamento, gpp.Producto, Cod_prod.Comprados_Prod
FROM Cod_prod
INNER JOIN `global.productos` gpp
ON cod_prod.Producto = Codigo
ORDER BY Cod_prod.Comprados_Prod DESC

WITH Cliente_compras AS (
  SELECT Numero, producto, COUNT(Producto) AS Total
  FROM `global.clientes`
  INNER JOIN `global.compras`
  ON Numero = cliente
  GROUP BY NUMERO, producto
),

QP_Cliente AS (
SELECT Numero, COUNT(Producto) AS cantidad_productos
FROM cliente_compras
GROUP BY Numero
)

SELECT Nombre_1, Nombre_2, Apellidos, QP_Cliente.cantidad_productos
FROM QP_Cliente
INNER JOIN `global.clientes`gcc
ON gcc.Numero = QP_Cliente.Numero
ORDER BY QP_Cliente.cantidad_productos DESC
